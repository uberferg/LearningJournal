Getting started: setting up the <canvas>

Like when using the canvas API, the first thing needed in order to use WebGL is an HTML5 <canvas> element. The code below creates one and sets onload='main()' in the body, which will be used to initialize our WebGL program. 

<body onload='main()'>
<canvas id='gl-canvas' width='500' height='500'>
  Oops, your browser does not support the canvas element.
</canvas>
</body>

The canvas must then be accessed and set up from wihin main(). First, we get the canvas using the document.getElementById() method. We then pass the reference to a function that will initialize the webGL context. In this function, the canvas' getContext() method is used to retrieve a WebGL context, using the experimental WebGL context as a fallback option.

If a context is successfully obtained, the color buffer is cleared to an RGBA color of our choosing, using the clearColor() and clear() methods. 

var gl;

function main() {
    var canvas = document.getElementById('gl-canvas');

    gl = initGLContext(canvas); // Initialize the GL canvas context

    // Only continue if WebGL is available
    if (gl) {
        gl.clearColor(1.0, 1.0, 1.0, 1.0);
        gl.clear(gl.COLOR_BUFFER_BIT);
    }
}

function initGLContext(canvas) {
    gl = null;

    try {
        gl = canvas.getContext('webgl') || 
             canvas.getContext('experimental-webgl');
    } catch(e) {}

    if (!gl) {
        alert("Unable to initialize WebGL. Your browser may not support it");
    }
    return gl;
}
